def fibonacci_serisi(n):
    # 3) diziyi oluştur
    fibonacci_sayisi = [0] * max(n, 2)

    # 4) ilk iki elemanı ata
    fibonacci_sayisi[0] = 0
    fibonacci_sayisi[1] = 1

    # 5) döngü sayacı
    index = 2

    # 6) index < N oldukça devam et
    while index < n:
        fibonacci_sayisi[index] = fibonacci_sayisi[index - 1] + fibonacci_sayisi[index - 2]
        index += 1

    # 7) ilk n terimi döndür
    return fibonacci_sayisi[:n]

int main(void) {

    // 3) diziyi olustur
    int fibonacci_sayisi[100];

    // 4) ilk iki elemani ata
    fibonacci_sayisi[0] = 0;
    fibonacci_sayisi[1] = 1;

    // 5) dongu sayaci
    int index = 2;

    // 6) index < N oldukca devam et
    while (index < n) {
        fibonacci_sayisi[index] = fibonacci_sayisi[index - 1] + fibonacci_sayisi[index - 2];
        index = index + 1;
    }

    // 7) ilk n terimi yazdir
    for (int i = 0; i < n; i++) {
        printf("%d", fibonacci_sayisi[i]);
        if (i < n - 1) printf(", ");
    }
    printf("\n");

    // 8) son
    return 0;
}

function fibonacciSerisi(n) {
  // 3) diziyi olustur
  const fibonacciSayisi = new Array(Math.max(n, 2)).fill(0);

  // 4) ilk iki elemani ata
  fibonacciSayisi[0] = 0;
  fibonacciSayisi[1] = 1;

  // 5) dongu sayaci
  let index = 2;

  // 6) index < N oldukca devam et
  while (index < n) {
    fibonacciSayisi[index] = fibonacciSayisi[index - 1] + fibonacciSayisi[index - 2];
    index += 1;
  }

  // 7) ilk n terimi dondur
  return fibonacciSayisi.slice(0, n);
}

5) Döngü sayacı olarak index = 2 ata.
6) Eğer index < N ise:
     - fibonacci_sayisi[index] = fibonacci_sayisi[index-1] + fibonacci_sayisi[index-2]
     - index = index + 1
     - Adım 6'ya geri dön.
7) Değilse: Dizinin elemanlarını baştan sona ekrana yazdır.
